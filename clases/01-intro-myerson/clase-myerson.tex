\frame {
  \frametitle{El enfoque institucional}

  \begin{align*}
    \text{instituciones} & = \text{reglas del juego} \\
                         & \neq \text{organizaciones}
  \end{align*}
\bigskip

\textbf{Analogía} \\
Las \alert{organizaciones} son equipos que se enfrentan en un juego cuyas reglas definen las \alert{instituciones} \\ 
\medskip \footnotesize{$\rightarrow$ P.ej. partidos, grupos de interés, legisladores...}
}
